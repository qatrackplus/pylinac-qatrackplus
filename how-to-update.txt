How I update the "thin" version of pylinac without machinelearning or demo files
git checkout mlc     # currently mlc branch has remote github.com/randlet/pylinac.git
git fetch upstream   # fetch James's latest branch
git merge upstream/<branch>   # mege in newest changes
git commit -am 'merge upstream<branch>'
git checkout master   # master of 'thin' repo (github.com/qatrackplus/pylinac-qatrackplus.git)
git checkout mlc -- *   # check out *all* files from mlc branch. This will overwrite all files on master
git reset HEAD
rm -rf machinelearning pylinac/demo_files  # delete demo files / machinelearning
git diff pylinac # review changes
git add <files>  # add changes to commit
git commit -a -m 'message'
